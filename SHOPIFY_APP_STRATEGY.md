   // lib/seo/rollback.ts
   async function rollbackFix(fixId: string) {
     const fix = await db.fix.findUnique({
       where: { id: fixId },
       include: { connection: true }
     });

     // Check rollback window
     if (isAfter(new Date(), fix.rollbackDeadline)) {
       throw new Error('Rollback window expired (90 days)');
     }

     // Restore previous state
     const previousState = JSON.parse(fix.beforeState);
     await shopify.updateProduct(previousState);

     // Mark as rolled back
     await db.fix.update({
       where: { id: fixId },
       data: {
         status: 'ROLLED_BACK',
         rolledBackAt: new Date()
       }
     });
   }
